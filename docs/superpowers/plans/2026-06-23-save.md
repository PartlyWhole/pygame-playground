# Save / download Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax.

**Goal:** A Save button that downloads a lone file as `.py`, or a multi-file/has-assets project as a self-contained `.zip` (code + assets), entirely client-side.

**Architecture:** Additive to `index.html`: a `#saveBtn`, a `saveProject()` dispatcher (lone-`.py` via Blob+`a[download]`; otherwise a lazily-loaded JSZip zip bundling `project.serialize().files` + `assetStore.getAll()` bytes), a `downloadBlob` helper, a lazy `loadJSZip`, and a `Cmd/Ctrl-S` editor binding. Design: `docs/specs/2026-06-23-save-design.md`. Proven spike: `test/spike-save.mjs`.

**Tech Stack:** Blob + `a[download]`, JSZip 3.10.1 (cdnjs, lazy), CodeMirror 5 keymap, Playwright download capture.

```js
// ---------------------------------------------------------------- save / download
const JSZIP_CDN = "https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js";
let _jszip = null;
function loadJSZip() {
  // Lazy: only fetched the first time a zip is produced, so the .py-only path
  // and first paint stay library-free.
  return _jszip ??= new Promise((resolve, reject) => {
    const s = document.createElement("script");
    s.src = JSZIP_CDN;
    s.onload = () => resolve(window.JSZip);
    s.onerror = () => { _jszip = null; reject(new Error("JSZip load failed")); };
    document.head.appendChild(s);
  });
}
function downloadBlob(data, filename, type = "application/octet-stream") {
  const blob = data instanceof Blob ? data : new Blob([data], { type });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url; a.download = filename;
  document.body.appendChild(a); a.click(); a.remove();
  setTimeout(() => URL.revokeObjectURL(url), 1000);
}
async function saveProject() {
  // Lone file with no uploaded assets -> a plain .py (no library needed).
  if (!project.isMulti() && assetFS.list.length === 0) {
    downloadBlob(project.text(project.entry), project.entry, "text/x-python");
    return;
  }
  // Otherwise a self-contained zip of all .py files + all assets.
  let JSZip;
  try { JSZip = await loadJSZip(); }
  catch { logLine("Couldn't load the zip library — check your connection.", "sys"); return; }
  const zip = new JSZip();
  const files = project.serialize().files;
  for (const name of Object.keys(files)) zip.file(name, files[name]);
  for (const rec of await assetStore.getAll()) zip.file(rec.name, new Uint8Array(rec.bytes));
  const blob = await zip.generateAsync({ type: "blob" });
  downloadBlob(blob, "pygame-project.zip");
}
document.getElementById("saveBtn").addEventListener("click", saveProject);
```

Add `Cmd-S`/`Ctrl-S` to the editor `extraKeys` (the object passed to `CodeMirror.fromTextArea`, ~line 752). Add these entries:

```js

// Helper: read a downloaded zip's entries via JSZip in-page.
async function readZip(download) {
  const path = join(tmp, 'p.zip');
  await download.saveAs(path);
  const b64 = (await readFile(path)).toString('base64');
  return page.evaluate(async (b64) => {
    const data = Uint8Array.from(atob(b64), c => c.charCodeAt(0));
    const Z = window.JSZip || (await new Promise((res, rej) => { const s = document.createElement('script'); s.src = 'https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js'; s.onload = () => res(window.JSZip); s.onerror = rej; document.head.appendChild(s); }));
    const zip = await Z.loadAsync(data);
    const out = {};
    for (const name of Object.keys(zip.files)) out[name] = Array.from(await zip.file(name).async('uint8array'));
    return out;
  }, b64);
}
const toStr = (arr) => Buffer.from(arr).toString('utf8');

// 6. Multi-file project -> pygame-project.zip with all .py files.
await page.evaluate(() => {
  window.project.load({ files: { 'main.py': 'import enemy\n', 'enemy.py': 'X = 9\n' },
                        order: ['main.py','enemy.py'], entry: 'main.py', active: 'main.py' });
});
const [zdl] = await Promise.all([ page.waitForEvent('download'), page.click('#saveBtn') ]);
if (zdl.suggestedFilename() === 'pygame-project.zip') ok('multi-file saves as pygame-project.zip');
else fail('zip name wrong: ' + zdl.suggestedFilename());
const z1 = await readZip(zdl);
if (toStr(z1['main.py']).includes('import enemy') && toStr(z1['enemy.py']).includes('X = 9'))
  ok('zip contains all .py files with correct content');
else fail('zip .py content wrong: ' + JSON.stringify(Object.keys(z1)));

// 7. Assets are bundled into the zip (upload a real PNG fixture, single file -> zip via the assets branch).
```
