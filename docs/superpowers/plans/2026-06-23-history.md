# History panel Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** A version-history panel: snapshots captured on each Run (local IndexedDB, both solo & room), a list, a line-diff of a past version vs now, and restore.

**Architecture:** Additive to `index.html`: a `historyStore` (its own IndexedDB DB), a fire-and-forget `captureSnapshot()` in `run()`, and a `🕘 History` popover (`renderHistory`/`showDiff`/`restoreSnapshot`) with a lazily-loaded `jsdiff`. No Automerge (solo path stays bundle-free). Design: `docs/specs/2026-06-23-history-design.md`. Proven: `test/spike-history.mjs` + a jsdiff no-build probe.

**Tech Stack:** IndexedDB (assetStore pattern), jsdiff 5.2.0 (cdnjs, lazy), the existing `project` model, Playwright.

In `index.html`, after `assetStore` (or near it), add:

```js
// ---------------------------------------------------------------- history (version snapshots)
// A snapshot of the whole project is saved on each Run, in a SEPARATE IndexedDB
// database so it can't disturb assetStore's versioning. Local-only; no Automerge.
const HISTORY_CAP = 100;
const historyStore = {
  _db: null,
  _open() {
    return this._db ??= new Promise((resolve, reject) => {
      const r = indexedDB.open('pygame-playground-history', 1);
      r.onupgradeneeded = () => r.result.createObjectStore('snapshots', { keyPath: 'id', autoIncrement: true });
      r.onsuccess = () => resolve(r.result);
      r.onerror = () => reject(r.error);
    });
  },
  async _tx(mode, fn) {
    const db = await this._open();
    return new Promise((resolve, reject) => {
      const tx = db.transaction('snapshots', mode);
      const out = fn(tx.objectStore('snapshots'));
      tx.oncomplete = () => resolve(out);
      tx.onerror = () => reject(tx.error);
    });
  },
  async getAll() {   // newest-first (autoincrement id ascending -> reverse)
    try { const all = await this._tx('readonly', s => { const rq = s.getAll(); return new Promise(r => rq.onsuccess = () => r(rq.result)); }); return all.reverse(); }
    catch { return []; }
  },
  async latest() { return (await this.getAll())[0] || null; },
  async add(rec) {
    try {
      await this._tx('readwrite', s => s.add(rec));
      const keys = await this._tx('readonly', s => { const rq = s.getAllKeys(); return new Promise(r => rq.onsuccess = () => r(rq.result)); });
      if (keys.length > HISTORY_CAP) { const drop = keys.slice(0, keys.length - HISTORY_CAP); await this._tx('readwrite', s => drop.forEach(k => s.delete(k))); }
    } catch {}
  },
  async clear() { try { await this._tx('readwrite', s => s.clear()); } catch {} },
};
window.historyStore = historyStore;   // test seam

async function captureSnapshot() {
  try {
    const proj = project.serialize();
    const last = await historyStore.latest();
    if (last && JSON.stringify(last.project) === JSON.stringify(proj)) return;   // dedup unchanged re-runs
    await historyStore.add({ at: Date.now(), mode: collab.active ? "room" : "solo", project: proj });
    if (typeof historyPanelEl !== "undefined" && !historyPanelEl.hidden) renderHistory();
  } catch {}
}
```

```js
// ---------------------------------------------------------------- history UI
const JSDIFF_CDN = "https://cdnjs.cloudflare.com/ajax/libs/jsdiff/5.2.0/diff.min.js";
let _diff = null;
function loadDiffLib() {
  return _diff ??= new Promise((resolve, reject) => {
    const s = document.createElement("script"); s.src = JSDIFF_CDN;
    s.onload = () => resolve(window.Diff); s.onerror = () => { _diff = null; reject(new Error("diff load")); };
    document.head.appendChild(s);
  });
}
const historyBtnEl = document.getElementById("historyBtn");
const historyPanelEl = document.getElementById("historyPanel");
const escHist = (s) => s.replace(/[&<>"]/g, c => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;" }[c]));
const lineCount = (t) => (t ? t.split("\n").length : 0);
function relTime(ms) {
  const s = Math.floor((Date.now() - ms) / 1000);
  if (s < 10) return "just now";
  if (s < 60) return s + "s ago";
  if (s < 3600) return Math.floor(s / 60) + "m ago";
  if (s < 86400) return Math.floor(s / 3600) + "h ago";
  return new Date(ms).toLocaleDateString();
}
function deltaBadge(older, cur) {
  const f = cur.entry;
  if (!older) return lineCount(cur.files[f]) + " lines";
  const d = lineCount(cur.files[f]) - lineCount(older.files[f] || "");
  return d > 0 ? "+" + d + " lines" : d < 0 ? d + " lines" : "edited";
}
let _histSnaps = [];
async function renderHistory() {
  if (historyPanelEl.hidden) return;
  _histSnaps = await historyStore.getAll();   // newest-first
  if (!_histSnaps.length) { historyPanelEl.innerHTML = '<div class="hp-head"><span>Version history</span></div><div class="hp-empty">No versions yet — press ▶ Run to save one.</div>'; return; }
  const rows = _histSnaps.map((s, i) => {
    const older = _histSnaps[i + 1];
    const nf = s.project.order.length;
    return `<div class="hist-row" data-id="${s.id}">
      <span class="hist-time">${relTime(s.at)}</span>
      <span class="hist-meta">${deltaBadge(older, s.project)}${nf > 1 ? " · " + nf + " files" : ""}${s.mode === "room" ? " · room" : ""}</span>
    </div>`;
  }).join("");
  historyPanelEl.innerHTML = `<div class="hp-head"><span>Version history</span><span class="hp-clear">Clear</span></div>${rows}<div class="hp-diff" id="histDiff"></div>`;
}
async function showDiff(id) {
  const rec = _histSnaps.find(s => String(s.id) === String(id));
  if (!rec) return;
  historyPanelEl.querySelectorAll(".hist-row").forEach(r => r.classList.toggle("sel", r.dataset.id === String(id)));
  const diffEl = document.getElementById("histDiff");
  diffEl.innerHTML = "Loading diff…";
  let Diff;
  try { Diff = await loadDiffLib(); } catch { diffEl.textContent = "Couldn't load the diff view — check your connection."; return; }
  const active = project.active;
  const before = rec.project.files[active] ?? "";
  const after = project.text(active) ?? "";
  const body = Diff.diffLines(before, after).map(p => {
    const cls = p.added ? "d-add" : p.removed ? "d-del" : "d-ctx";
    return p.value.replace(/\n$/, "").split("\n").map(ln => `<div class="${cls}">${escHist(ln) || "&nbsp;"}</div>`).join("");
  }).join("");
  const otherChanged = [...new Set([...Object.keys(rec.project.files), ...project.order])]
    .filter(n => n !== active && (rec.project.files[n] ?? "") !== (project.files[n]?.getValue() ?? "")).length;
  diffEl.innerHTML = `<div class="hp-difftitle">${escHist(active)} — this version → now${otherChanged ? ` (+${otherChanged} other file${otherChanged > 1 ? "s" : ""} differ)` : ""}</div>`
    + `<div class="hp-diffbody">${body || '<div class="d-ctx">(no changes in this file)</div>'}</div>`
    + `<button class="hp-restore" data-id="${rec.id}">Restore this version</button>`;
}
function restoreSnapshot(id) {
  const rec = _histSnaps.find(s => String(s.id) === String(id));
  if (!rec) return;
  if (!confirm("Replace your current code with this version?")) return;
  project.load(rec.project);
  renderTabs();
  flushSave();
  historyPanelEl.hidden = true;
}
historyBtnEl.addEventListener("click", () => { historyPanelEl.hidden = !historyPanelEl.hidden; renderHistory(); });
historyPanelEl.addEventListener("click", (e) => {
  if (e.target.classList.contains("hp-clear")) { historyStore.clear().then(renderHistory); return; }
  const restore = e.target.closest(".hp-restore");
  if (restore) { restoreSnapshot(restore.dataset.id); return; }
  const row = e.target.closest(".hist-row");
  if (row) showDiff(row.dataset.id);
});
document.addEventListener("click", (e) => {
  if (!historyPanelEl.hidden && !historyPanelEl.contains(e.target) && e.target !== historyBtnEl) historyPanelEl.hidden = true;
});
```
